
Q1) WAQTD NAME OF THE EMPLOYEES ALONG WITH THEIR ANNUAL SALARY.
--->
SELECT ENAME, SAL*12 ANNUALSAL
FROM EMP;

Q2) WAQTD ENAME AND JOB FOR ALL THE EMPLOYEES WITH THEIR HALF TERM SALARY.
--->
SELECT ENAME, JOB, SAL*6 HALF_TERM_SAL
FROM EMP;

Q3) WAQTD ALL THE DETAILS OF THE EMPLOYEES ALONG WITH AN ANNUAL BONUS OF 2000.
--->
SELECT EMP.*, SAL*12+2000 
FROM EMP;

Q4) WAQTD NAME, SALARY AND SALARY WITH A HIKE OF 10%.
--->
SELECT ENAME, SAL, SAL+(SAL*10/100)
FROM EMP;

Q5) WAQTD NAME AND SALARY WITH DEDUCTION OF 25%.
--->
SELECT ENAME, SAL-(SAL*25/100)
FROM EMP;

Q6) WAQTD NAME AND SALARY WITH MONTHLY HIKE OF 50.
--->
SELECT ENAME, SAL+50
FROM EMP;

Q7) WAQTD NAME AND ANNUAL SALARY WITH DEDUCTION OF 10%?
--->
SELECT ENAME, (SAL*12) –((SAL*12)/100)*10
FROM EMP;

Q8) WAQTD TOTAL SALARY GIVEN TO EACH EMPLOYEE (SALARY + COMMISSION).
--->
UPDATE EMP
SET COMM=0
WHERE COMM IS NULL;

SELECT SAL+COMM
FROM EMP;

Q9) WAQTD DETAILS OF THE EMPLOYEES ALONG WITH ANNUAL SALARY.
--->
SELECT EMP.*, SAL*12 “ANNUAL SAL”
FROM EMP;

Q10) WAQTD NAME AND DESIGNATION ALONG WITH 100 PENALTY IN SALARY.
--->
SELECT ENAME, JOB, SAL-100
FROM EMP;
